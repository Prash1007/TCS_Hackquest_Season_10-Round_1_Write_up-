// CHALLENGE 01
Seeds of Time
Crypto
FLAG HQX{c126bb454bc2652acea1baf0729fbd08}
Approach
  • 1Converted the ciphertext from hex to bytes using binascii.unhexlify().
  • 2Bruteforced possible timestamps in the range 1765584000 → 1765670399 UTC — a 24‑hour window around the challenge release date.
  • 3For each seed, generated a keystream with random.random() * 256, XOR'd against the cipher, and matched the b'HQX{' prefix to identify the correct seed.
PYTHON · sc.py 
import random
import time
from binascii import unhexlify

cipher_hex = "f60d1ef6307bc56ed4f3f8fe41eacccf86b820fcdd8e9dd41eb0083b601303b923ebba8..."
cipher  = unhexlify(cipher_hex)
length  = len(cipher)

NOW = int(time.time())

for seed in range(NOW - 200000, NOW + 200000):
    random.seed(seed)
    keystream  = bytes(int(random.random() * 256) for _ in range(length))
    plaintext  = bytes(c ^ k for c, k in zip(cipher, keystream))
    try:
        text = plaintext.decode()
        if text.startswith("HQX{") and text.endswith("}"):
            print("[+] Seed:", seed)
            print("[+] FLAG:", text)
            break
    except:
        pass

PYTHON · lsb.py 
from PIL import Image
import numpy as np

img = Image.open("image_dCA9d30958.png")
arr = np.array(img)
bits = []

for y in range(arr.shape[0]):
    for x in range(arr.shape[1]):
        for ch in range(3):   # R, G, B
            bits.append(arr[y, x, ch] & 1)

data = []
for i in range(0, len(bits) - 7, 8):
    byte = 0
    for j in range(8):
        byte = (byte << 1) | bits[i + j]
    data.append(byte)

message = bytes(data)
print(message)

// CHALLENGE 05
Address Abyss
Misc
FLAG HQX{e1f63411d09fa0eb766745bb0860d9f4}
Approach
  • 1Given a large file of mixed IPv4 / IPv6 addresses. The hint: IPv4: 92.7.X.Y encodes index=X, char=Y; IPv6: 2510:a1:X::Y encodes index=hex(X), char=Y.
  • 2Example — 92.7.2.0 → index 2, char '0'  |  2510:a1:b::f → index 11 (0xb), char 'f'.
  • 3Wrote a script using re.fullmatch() to parse both patterns, build a dict of {index: char}, then sort by index to assemble the flag.
PYTHON · sc.py 
import re

flag = {}
with open("ip_logs_2a669E4705.txt", "r", errors="ignore") as f:
    for line in f:
        line = line.strip()
        # IPv4 pattern
        m4 = re.fullmatch(r"92\.7\.(\d+)\.(\d+)", line)
        if m4:
            idx, val = int(m4.group(1)), m4.group(2)
            if len(val) == 1: flag[idx] = val
            continue
        # IPv6 pattern
        m6 = re.fullmatch(r"2510:a1:([0-9a-fA-F]+)::([0-9a-fA-F])", line)
        if m6:
            idx = int(m6.group(1), 16)
            flag[idx] = m6.group(2)

flag_str = ""
for i in sorted(flag): flag_str += flag[i]
print("HQX{" + flag_str + "}")
